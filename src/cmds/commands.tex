% param 1 italian
% param 2 albanian
\NewDocumentCommand{\addTranslation}{O{#3} O{#4} m m}{%
    % FORM ITALIAN TO ALBANIAN
    \newglossaryentry{#1}%
    {%
        type={itan},%
        name={#3},%
        description={#4}%
    }%
    %
    % FORM ALBANIAN TO ITALIAN
    \newglossaryentry{#2}%
    {%
        type={anit},%
        name={#4},%
        description={#3}%
    }%
}

\NewDocumentCommand{\addTranslationRow}{O{#2} m}{%

}
